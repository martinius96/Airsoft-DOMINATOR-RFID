Descrizione progetto punti WiFi:

Modalità di gioco Cattura punti per gli sport Airsoft e Paintball. I punti comunicano tra loro tramite moduli WiFi della serie nRF24L01, mentre si scambiano informazioni sullo stato del punto (se è occupato e da quale squadra). Un punto è un trasmettitore permanente e richiede la comunicazione dall'altro punto. Il secondo punto è il ricevitore e risponde ai dati ricevuti dal trasmettitore, evitando così collisioni durante la comunicazione sulla frequenza WiFi 2.4 GHz. Ciascuno dei punti utilizza un microcontrollore di controllo Arduino (Arduino Uno / Nano) o un chip separato (Atmel) AtMega328P, che controlla l'intera logica del sistema. Ci sono 2 squadre nel gioco - RED Squadra un GRE Squadra, che giocano l'uno contro l'altro e cercano di occupare entrambi i punti premendo l'apposito pulsante. L'inserimento del pulsante indica l'azione di occupare un punto da parte di un membro del team RED o GRE. Vengono utilizzati i pulsanti collegati in modalità INPUT_PULLUP. Premendo il pulsante si invia il segnale GND al terminale di ingresso digitale di Arduino (segnale di commutazione Active-LOW). La durata della pressa / numero di pressioni non è controllata e non è configurabile. Il punto risponde immediatamente alla pressione di un pulsante. Il gioco finisce quando entrambi i punti sono occupati dalla stessa squadra. La variazione del punto è indicata dal rilevatore, alla fine del gioco entrambi i rilevatori emettono un bip in un loop infinito e contemporaneamente viene interrotta la comunicazione WiFi tra i moduli nRF24L01. I moduli WiFi nRF24l01 comunicano a 2,4 GHz. nRF24l01 (senza + PA + LNA) comunicano a circa 70 metri, nRF24l01 + PA + LNA anche per chilometro, rispettivamente in foresta e ambiente altrimenti disturbato a quota 450-700 metri alla massima potenza di trasmissione. Per l'alimentazione esterna, è necessario utilizzare un adattatore YL con un convertitore da 3,3 V e alimentarlo tramite un adattatore esterno. Non utilizzare un alimentatore 5V da Arduino, c'è il rischio di distruggere il controller, così come il terminale 3V3 non è adatto per alimentare il modulo, in quanto ha un elevato consumo di corrente nelle modalità di trasmissione più potenti. Il riavvio di entrambi i punti viene eseguito ripristinando il pulsante Arduín o estraendo e collegando l'alimentatore (ciclo di accensione).

I moduli wireless comunicano a 2,4 GHz, non è una tecnologia WiFi, non richiedono un router o un punto di accesso! I moduli hanno il proprio canale di comunicazione con una scelta fino a 140 canali.

    Programma di test - punti WiFi

    Lo scopo del programma di test è testare la compatibilità dell'hardware (versioni dei moduli nRF24L01 con una velocità di trasferimento di 250kbps, poiché non tutti lo supportano). Testare e implementare una connessione tra il trasmettitore e il ricevitore con una richiamata del ricevitore, che restituisce i dati inviati al trasmettitore. Il ricevitore ha un cicalino integrato che emette un segnale acustico quando rileva determinati dati: indica che c'è una connessione al trasmettitore e che i dati stanno fluendo. Il tester è adatto anche per l'utilizzo nel gioco Airsoft per il test iniziale della portata e della comunicazione tra i due nodi. Il trasmettitore trasmette sempre, il ricevitore solo in risposta al trasmettitore dopo aver ricevuto i dati. È possibile trasmettere dati diversi nella struttura, pacchetti più grandi.

    Tester del trasmettitore (Questo non è un programma per il progetto Airsoft WiFi! - Il progetto è pagato)

    //nRF24L01 + PA + LNA - Transmitter tester
//Author: Martin Chlebovec (martinius96)
//Web: https://arduino.php5.sk
//Donate: https://paypal.me/chlebovec

#include <SPI.h>
#include "RF24.h"
// nastavení propojovacích pinů
#define CE 4
#define CS 3
// inicializace nRF s piny CE a CS
RF24 nRF(CE, CS);
// nastavení adres pro přijímač a vysílač,
// musí být nastaveny stejně v obou programech!
byte adresaPrijimac[]= "prijimac00";
byte adresaVysilac[]= "vysilac00";
void setup() {
  // komunikace přes sériovou linku rychlostí 9600 baud
  Serial.begin(9600);
  // zapnutí komunikace nRF modulu
  nRF.begin();
  nRF.setDataRate( RF24_250KBPS );
  // nastavení výkonu nRF modulu,
  // možnosti jsou RF24_PA_MIN, RF24_PA_LOW, RF24_PA_HIGH and RF24_PA_MAX,
  // pro HIGH a MAX je nutnýv externí 3,3V zdroj
  nRF.setPALevel(RF24_PA_LOW);
  // nastavení zapisovacího a čtecího kanálu
  nRF.openWritingPipe(adresaVysilac);
  nRF.openReadingPipe(1,adresaPrijimac);
  // začátek příjmu dat
  nRF.startListening();
}

void loop() {
  // for smyčka pro postupné odeslání
  // hodnot 0 až 3 pro načtení všech dat
  // z přijímače
  for (int i=0; i < 4; i++ ) {
    // ukončení příjmu dat
    nRF.stopListening();
    // vytisknutí aktuální volby po sériové lince
    Serial.print("Posilam volbu ");
    Serial.println(i);
    // uložení startovního času komunikace
    unsigned long casZacatek = micros();
    // odeslání aktuální volby, v případě selhání
    // vytištění chybové hlášky na sériovou linku
    if (!nRF.write( &i, sizeof(i) )){
       Serial.println("Chyba při odeslání!");
    }
    nRF.startListening();
    unsigned long casCekaniOdezvy = micros();
    boolean timeout = false;
    while ( ! nRF.available() ){
    if (micros() - casCekaniOdezvy > 200000 ){
          timeout = true;
          break;
      }      
    }
    if ( timeout ){
      Serial.println("Chyba při prijmu, vyprseni casu na odezvu!");
    }
    else{
     
        unsigned long prijataData;
        nRF.read( &prijataData, sizeof(prijataData) );
        unsigned long casKonec = micros();
        Serial.print("Odeslana volba: ");
        Serial.print(i);
        Serial.print(", prijata data: ");
        Serial.println(prijataData);
        Serial.print("Delka spojeni: ");
        Serial.print(casKonec - casZacatek);
        Serial.println(" mikrosekund.");
    }
    delay(200);
  }
}

    Tester ricevitore (Questo non è un programma per il progetto Airsoft WiFi! - Il progetto è pagato)

    //nRF24L01 + PA + LNA - Receiver tester
//Author: Martin Chlebovec (martinius96)
//Web: https://arduino.php5.sk
//Donate: https://paypal.me/chlebovec

#include <SPI.h>
#include "RF24.h"
int prijem;
const int buzzer = 2;
#define CE 4
#define CS 3
RF24 nRF(CE, CS);
byte adresaPrijimac[] = "prijimac00";
byte adresaVysilac[] = "vysilac00";
void setup() {
  Serial.begin(9600);
  pinMode(buzzer, OUTPUT);
  nRF.begin();
  nRF.setDataRate( RF24_250KBPS );
  nRF.setPALevel(RF24_PA_LOW);
  nRF.openWritingPipe(adresaPrijimac);
  nRF.openReadingPipe(1, adresaVysilac);
  nRF.startListening();
}

void loop() {
  unsigned long odezva;
  if ( nRF.available()) {
    // čekání na příjem dat
    while (nRF.available()) {
      nRF.read( &prijem, sizeof(prijem) );
    }
    Serial.print("Prijata volba: ");
    Serial.print(prijem);
    switch ( prijem ) {
      case 1:

        odezva = millis();

        tone(buzzer, 50, 500);

        break;
      case 2:
        odezva = millis() / 1000;
        break;
      case 3:
        odezva = micros();
        break;
      default:
        odezva = 0;
        break;
    }
    nRF.stopListening();
    nRF.write( &odezva, sizeof(odezva) );
    nRF.startListening();
  }
}
